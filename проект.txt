<!-- ===== JAVASCRIPT ===== -->
<script>

    /* ── ДАННЫЕ ЧЕКЛИСТА ─────────────────────────── */
    const ITEMS = [
        { id: 'teeth',     icon: '🦷', text: 'Почистить зубы',                  time: 'Утро & Вечер' },
        { id: 'face',      icon: '💧', text: 'Умыться',                          time: 'Утро & Вечер' },
        { id: 'shower',    icon: '🚿', text: 'Принять душ',                      time: 'Ежедневно'    },
        { id: 'hair',      icon: '💆', text: 'Расчесать волосы',                 time: 'Утром'        },
        { id: 'hands',     icon: '✋', text: 'Вымыть руки перед едой',           time: 'Каждый раз'   },
        { id: 'deodorant', icon: '🌬️', text: 'Использовать дезодорант',          time: 'После душа'   },
        { id: 'clothes',   icon: '👕', text: 'Сменить нижнее бельё и носки',     time: 'Ежедневно'    },
        { id: 'nails',     icon: '💅', text: 'Очистить под ногтями',             time: 'Ежедневно'    }
    ];

    /* ── СОСТОЯНИЕ ───────────────────────────────── */
    let checked      = new Set();
    let todayDone    = false;

    /* ── УТИЛИТЫ ─────────────────────────────────── */
    const todayStr = () => new Date().toISOString().split('T')[0];

    const yesterdayStr = () => {
        const d = new Date();
        d.setDate(d.getDate() - 1);
        return d.toISOString().split('T')[0];
    };

    const daysWord = n => {
        const a = Math.abs(n);
        if (a % 100 >= 11 && a % 100 <= 19) return 'дней';
        if (a % 10 === 1)                    return 'день';
        if (a % 10 >= 2 && a % 10 <= 4)     return 'дня';
        return 'дней';
    };

    /* ── LS HELPERS ──────────────────────────────── */
    const lsGet  = (k, def) => { try { return JSON.parse(localStorage.getItem(k)) ?? def; } catch { return def; } };
    const lsSet  = (k, v)   => localStorage.setItem(k, JSON.stringify(v));

    /* ── ИНИЦИАЛИЗАЦИЯ ───────────────────────────── */
    function init() {
        renderDate();
        checkStreakBreak();
        loadTodayState();
        renderChecklist();
        updateProgress();
        updateStreakUI();
        renderCalendar();
    }

    function renderDate() {
        const opts = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
        document.getElementById('currentDate').textContent =
            '📅 ' + new Date().toLocaleDateString('ru-RU', opts);
    }

    /* Проверяем: если пропущен хоть 1 день — сбрасываем серию */
    function checkStreakBreak() {
        const last = lsGet('hy_lastDate', '');
        const today = todayStr();
        const yest  = yesterdayStr();

        if (last && last !== today && last !== yest) {
            lsSet('hy_streak', 0);
        }
    }

    function loadTodayState() {
        const last = lsGet('hy_lastDate', '');
        todayDone = (last === todayStr());

        if (todayDone) {
            checked = new Set(lsGet('hy_todayChecked', []));
        }
    }

    /* ── РЕНДЕР ЧЕКЛИСТА ─────────────────────────── */
    function renderChecklist() {
        const wrap = document.getElementById('checklistWrap');
        wrap.innerHTML = '';

        ITEMS.forEach(item => {
            const div = document.createElement('div');
            div.className = 'checklist-item' + (checked.has(item.id) ? ' checked' : '');
            div.id = 'ci-' + item.id;
            div.setAttribute('role', 'checkbox');
            div.setAttribute('aria-checked', checked.has(item.id));
            div.onclick = () => toggleItem(item.id);

            div.innerHTML = `
                <div class="checkbox-box">${checked.has(item.id) ? '✓' : ''}</div>
                <span class="item-icon">${item.icon}</span>
                <span class="item-text">${item.text}</span>
                <span class="item-time">${item.time}</span>
            `;
            wrap.appendChild(div);
        });

        if (todayDone) {
            wrap.querySelectorAll('.checklist-item').forEach(el => el.style.cursor = 'default');
            showDoneMsg();
            const btn = document.getElementById('completeBtn');
            btn.disabled = true;
            btn.innerHTML = '✅ Сегодня уже выполнено!';
        }
    }

    function toggleItem(id) {
        if (todayDone) return;

        checked.has(id) ? checked.delete(id) : checked.add(id);

        const el = document.getElementById('ci-' + id);
        const isNow = checked.has(id);
        el.className = 'checklist-item' + (isNow ? ' checked' : '');
        el.querySelector('.checkbox-box').textContent = isNow ? '✓' : '';
        el.setAttribute('aria-checked', isNow);

        lsSet('hy_todayChecked', [...checked]);
        updateProgress();
    }

    function updateProgress() {
        const done  = checked.size;
        const total = ITEMS.length;
        const pct   = Math.round((done / total) * 100);

        document.getElementById('progressText').textContent = `${done} / ${total}`;
        document.getElementById('progressFill').style.width = pct + '%';

        const banner = document.getElementById('allDoneBanner');
        if (done === total && !todayDone) {
            banner.classList.add('show');
        } else {
            banner.classList.remove('show');
        }
    }

    /* ── ЗАВЕРШЕНИЕ ДНЯ ──────────────────────────── */
    function completeToday() {
        if (todayDone) return;

        // Помечаем всё как выполненное
        ITEMS.forEach(i => checked.add(i.id));

        const today = todayStr();
        const yest  = yesterdayStr();
        const last  = lsGet('hy_lastDate', '');
        let streak  = lsGet('hy_streak', 0);
        let best    = lsGet('hy_best', 0);

        streak = (last === yest) ? streak + 1 : 1;

        if (streak > best) { best = streak; lsSet('hy_best', best); }

        lsSet('hy_streak',      streak);
        lsSet('hy_lastDate',    today);
        lsSet('hy_todayChecked', [...checked]);

        // История
        let hist = lsGet('hy_history', []);
        if (!hist.includes(today)) {
            hist.push(today);
            if (hist.length > 30) hist = hist.slice(-30);
            lsSet('hy_history', hist);
        }

        todayDone = true;

        // Обновляем UI
        document.querySelectorAll('.checklist-item').forEach(el => {
            el.className = 'checklist-item checked';
            el.querySelector('.checkbox-box').textContent = '✓';
            el.style.cursor = 'default';
        });

        document.getElementById('allDoneBanner').classList.remove('show');
        updateProgress();
        updateStreakUI();
        renderCalendar();
        showDoneMsg();

        const btn = document.getElementById('completeBtn');
        btn.disabled = true;
        btn.innerHTML = '✅ Сегодня уже выполнено!';

        // Анимация карточки серии
        const card = document.getElementById('streakCard');
        card.classList.add('bounce-anim');
        setTimeout(() => card.classList.remove('bounce-anim'), 700);
    }

    function showDoneMsg() {
        const streak = lsGet('hy_streak', 0);
        document.getElementById('todayDoneWrap').innerHTML = `
            <div class="done-today-msg">
                <h3>🎉 Отлично! Сегодня выполнено!</h3>
                <p>Ты поддерживаешь серию уже <strong>${streak} ${daysWord(streak)}</strong> подряд!</p>
            </div>
        `;
    }

    /* ── СЕРИЯ ───────────────────────────────────── */
    function updateStreakUI() {
        const streak = lsGet('hy_streak', 0);
        const best   = lsGet('hy_best',   0);

        document.getElementById('streakNumber').textContent = streak;
        document.getElementById('bestLabel').textContent    = `🏆 Лучшая серия: ${best} ${daysWord(best)}`;

        const emojis = ['💤','🌱','🌱','🔥','🔥','🔥','🔥','⚡','⚡','⚡','⚡','⚡','⚡','⚡','🏆'];
        const idx = Math.min(streak, emojis.length - 1);
        document.getElementById('streakEmoji').textContent = emojis[idx];
    }

    function renderCalendar() {
        const hist = lsGet('hy_history', []);
        const today = todayStr();
        const grid  = document.getElementById('calGrid');
        grid.innerHTML = '';

        for (let i = 13; i >= 0; i--) {
            const d = new Date();
            d.setDate(d.getDate() - i);
            const ds = d.toISOString().split('T')[0];
            const day = d.getDate();

            const el = document.createElement('div');
            el.className = 'cal-day';
            if (hist.includes(ds))  el.classList.add('done');
            if (ds === today)       el.classList.add('today-mark');
            el.textContent = day;
            el.title = ds;
            grid.appendChild(el);
        }
    }

    /* ── МОДАЛКИ ─────────────────────────────────── */
    function openModal(id) {
        document.getElementById(id).classList.add('active');
        document.body.style.overflow = 'hidden';
    }

    function closeModal(id) {
        document.getElementById(id).classList.remove('active');
        document.body.style.overflow = '';
    }

    function outsideClose(e, id) {
        if (e.target === document.getElementById(id)) closeModal(id);
    }

    document.addEventListener('keydown', e => {
        if (e.key === 'Escape') {
            closeModal('howModal');
            closeModal('tipsModal');
        }
    });

    /* ── СТАРТ ───────────────────────────────────── */
    init();

</script>
